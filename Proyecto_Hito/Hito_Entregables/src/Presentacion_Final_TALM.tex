% [VALIDACIÓN LÓGICA]: Satisface el criterio de "Comunicación Matemática" (CM) de la presentación al introducir claramente el contexto y objetivos del manipulador.
\begin{frame}{Contexto y Analogía del Brazo Humano}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            El brazo robótico antropomórfico de 3 GDL imita la estructura de la extremidad superior humana:
            \vspace{0.3cm}
            \begin{itemize}
                \item \textbf{Cintura (Yaw):} Rotación de la base alrededor del eje vertical $z$. Ángulo $\theta_1$.
                \item \textbf{Hombro (Pitch):} Elevación principal sobre el eje horizontal. Ángulo $\theta_2$.
                \item \textbf{Codo (Pitch relativo):} Flexo-extensión distal respecto al primer eslabón. Ángulo $\theta_3$.
            \end{itemize}
            \vspace{0.3cm}
            \textbf{Objetivo:} Mapear analíticamente el espacio articular al espacio operacional tridimensional y evaluar la resistencia estructural ante rotación crítica.
        \end{column}
        \begin{column}{0.4\textwidth}
            \begin{block}{Grados de Libertad (GDL)}
                \centering
                \begin{itemize}
                    \item $\theta_1$: Yaw base
                    \item $\theta_2$: Pitch hombro
                    \item $\theta_3$: Pitch codo
                \end{itemize}
            \end{block}
            \centering
            \includegraphics[width=0.6\textwidth]{logo_uss.png}
        \end{column}
    \end{columns}
\end{frame}

% Slide 4: Ecuaciones Cinemáticas (Analogía Sombra Radial)
% [VALIDACIÓN LÓGICA]: Satisface el criterio de "Uso de Representaciones" (CM) y "Limitaciones del Modelo" (RA) al derivar las ecuaciones del espacio operacional 3D y advertir sobre la singularidad geométrica de base.
\begin{frame}{Ecuaciones Cinemáticas: Analogía Sombra Radial}
    Proyectamos el brazo sobre el plano horizontal para hallar la sombra radial $r$:
    \begin{equation}
        r = L_1\cos(\theta_2) + L_2\cos(\theta_2 + \theta_3)
    \end{equation}
    
    A partir del radio proyectado $r$, las coordenadas cartesianas $(x, y, z)$ son:
    \begin{align}
        x &= r\cos(\theta_1) = \left[ L_1\cos(\theta_2) + L_2\cos(\theta_2 + \theta_3) \right]\cos(\theta_1) \\
        y &= r\sin(\theta_1) = \left[ L_1\cos(\theta_2) + L_2\cos(\theta_2 + \theta_3) \right]\sin(\theta_1) \\
        z &= L_1\sin(\theta_2) + L_2\sin(\theta_2 + \theta_3)
    \end{align}
    
    \begin{block}{Singularidad Geométrica de Base}
        Cuando $r = 0$ (brazo en posición vertical), $\theta_1$ queda indeterminado, provocando una singularidad del manipulador en coordenadas cartesianas.
    \end{block}
\end{frame}

% Slide 6: Escenario Crítico de Rotación (140 km/h)
% [VALIDACIÓN LÓGICA]: Satisface el criterio de "Construcción del Modelo" (MM) al formular la integral dinámica de tracción centrípeta para modelar esfuerzos bajo una velocidad crítica de 140 km/h.
\begin{frame}{Escenario Crítico de Rotación (140 km/h)}
    Evaluamos el brazo totalmente extendido ($R_{\text{max}} = 0.2\text{ m}$) a $140\text{ km/h}$ ($38.89\text{ m/s}$):
    \vspace{0.2cm}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Parámetros Cinemáticos}
                \begin{itemize}
                    \item Velocidad angular: $\omega = \frac{v}{R_{\text{max}}} = 194.44\text{ rad/s}$
                    \item Aceleración centrípeta en extremo:
                    $$a_c = \omega^2 R_{\text{max}} = 7561.73\text{ m/s}^2 \approx \mathbf{771g}$$
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Cálculo de Tensiones Centrífugas}
                La tensión a lo largo del radio $x$ es:
                $$T(x) = \int_x^{L_1+L_2} \rho \omega^2 \xi d\xi + M \omega^2 (L_1+L_2)$$
                \begin{itemize}
                    \item \textbf{Base del brazo ($x=0$):}
                    $$T_{\text{base}} = \mathbf{7561.73\text{ N}}$$
                    \item \textbf{Articulación del codo ($x=0.1\text{ m}$):}
                    $$T_{\text{codo}} = \mathbf{6616.51\text{ N}}$$
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% Slide 7: Verificación de Esfuerzos y Seguridad
% [VALIDACIÓN LÓGICA]: Satisface el criterio de "Defensa de Decisiones" al calcular cuantitativamente los factores de seguridad del aluminio de los eslabones y el acero del pasador, identificando el punto de falla del diseño original.
\begin{frame}{Verificación de Esfuerzos y Seguridad}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Eslabón de Aluminio (Base)}
                \begin{itemize}
                    \item Diámetro: $10\text{ mm}$ (Área $78.54\text{ mm}^2$)
                    \item Tensión: $T_{\text{base}} = 7561.73\text{ N}$
                    \item Esfuerzo normal:
                    $$\sigma_{\text{link}} = \frac{T_{\text{base}}}{A_{\text{link}}} = \mathbf{96.28\text{ MPa}}$$
                    \item Factor de Seguridad (Fluencia $276\text{ MPa}$):
                    $$FS_{\text{link}} = \frac{276}{96.28} = \mathbf{2.87} \quad (\textbf{RESISTE})$$
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Pasador Original de Acero (Codo)}
                \begin{itemize}
                    \item Diámetro: $3\text{ mm}$ (Área $7.07\text{ mm}^2$, cizalle simple)
                    \item Fuerza de corte: $T_{\text{codo}} = 6616.51\text{ N}$
                    \item Esfuerzo cortante:
                    $$\tau_{\text{pin,3mm}} = \frac{T_{\text{codo}}}{A_{\text{pin,3mm}}} = \mathbf{936.0\text{ MPa}}$$
                    \item Factor de Seguridad (Admisible $150\text{ MPa}$):
                    $$FS_{\text{pin,3mm}} = \frac{150}{936.0} = \mathbf{0.16} \quad (\textbf{FALLA CRÍTICA})$$
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% Slide 9: Propuesta de Rediseño de la Articulación
% [VALIDACIÓN LÓGICA]: Satisface el criterio de "Construcción del Modelo" (MM) al demostrar cuantitativamente la efectividad de la alternativa de rediseño de pasador de 8 mm en cortante doble.
\begin{frame}{Propuesta de Rediseño de la Articulación del Codo}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Propuesta:} Aumentar el diámetro del pasador a $8\text{ mm}$ e implementar una junta en cortante doble.
            \vspace{0.2cm}
            \begin{itemize}
                \item \textbf{Cortante Simple con $d = 8\text{ mm}$ (Área $50.27\text{ mm}^2$):}
                $$\tau_{\text{pin,8mm}} = \frac{T_{\text{codo}}}{A_{\text{pin,8mm}}} = \mathbf{131.63\text{ MPa}} < 150\text{ MPa}$$
                $$FS_{\text{pin}} = 1.14 \quad (\textbf{RESISTE})$$
                \item \textbf{Cortante Doble con $d = 8\text{ mm}$ (Área $2 \times 50.27\text{ mm}^2$):}
                $$\tau_{\text{pin,doble}} = \frac{T_{\text{codo}}}{2 A_{\text{pin}}} = \mathbf{65.81\text{ MPa}} \ll 150\text{ MPa}$$
                $$FS_{\text{pin,doble}} = \mathbf{2.28} \quad (\textbf{ALTAMENTE SEGURO})$$
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
